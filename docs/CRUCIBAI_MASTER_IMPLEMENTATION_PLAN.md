```python
class ReActAgent:
    async def solve(self, problem):
        for step in range(max_steps):
            # Think
            thought = await self.reason(problem)
            
            # Act
            action = await self.execute(thought)
            
            # Observe
            result = await self.verify(action)
            
            # Check completion
            if await self.is_complete(result):
                return result
```
